CREATE TABLE IF NOT EXISTS PlayerStats (
    SteamId BIGINT UNSIGNED NOT NULL,
    Name VARCHAR(255) NULL,
    Kills INT NOT NULL DEFAULT 0,
    Headshots INT NOT NULL DEFAULT 0,
    PVPDeaths INT NOT NULL DEFAULT 0,
    PVEDeaths INT NOT NULL DEFAULT 0,
    Zombies INT NOT NULL DEFAULT 0,
    MegaZombies INT NOT NULL DEFAULT 0,
    Animals INT NOT NULL DEFAULT 0,
    Resources INT NOT NULL DEFAULT 0,
    Harvests INT NOT NULL DEFAULT 0,
    Fish INT NOT NULL DEFAULT 0,
    Structures INT NOT NULL DEFAULT 0,
    Barricades INT NOT NULL DEFAULT 0,
    Playtime BIGINT UNSIGNED NOT NULL DEFAULT 0,
    UIDisabled BOOLEAN NULL,
    LastUpdated DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT PK_PlayerStats_SteamId PRIMARY KEY (SteamId)
);
